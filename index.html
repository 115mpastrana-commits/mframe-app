function getCurveParams(){const{anchorLowTSS:lt,anchorLowRatio:lr,anchorHighTSS:ht,anchorHighRatio:hr}=curveConfig;const b=Math.log(hr/lr)/Math.log(ht/lt);const a=lr/Math.pow(lt,b);return{a,b};}

function getCurveRatio(tss){const{a,b}=getCurveParams();return a*Math.pow(tss,b);}

function getDotZone(tss,ratio){const optimal=getCurveRatio(tss),dev=(ratio-optimal)/optimal;if(Math.abs(dev)<=0.20)return'on';return dev<-0.20?'below':'above';}
